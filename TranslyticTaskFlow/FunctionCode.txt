import fabric.functions as fn
import logging
from datetime import datetime
from dateutil.parser import parse

date_str = "2025-06-08"


udf = fn.UserDataFunctions()

@udf.connection(argName="sqlDB",alias="contextDB")
@udf.function()
def addcomment(sqlDB: fn.FabricSqlConnection,salesday: str, comment: str) -> bool:
    logging.info('Python UDF trigger function processed a request.')

    dt = parse(salesday)  # parse string to datetime
    d = dt.date()  # extract the date part only

    data = (d,comment)

    # Insert data into the table
    insert_query = "INSERT INTO comments (salesdate,explanation) VALUES (?, ?);"

    # Establish a connection to the SQL database
    connection = sqlDB.connect()
    cursor = connection.cursor()
    cursor.execute(insert_query, data)

    # Commit the transaction
    connection.commit()

    # Close the connection
    cursor.close()
    connection.close()  

    return True
